| --------: | ------------------------------------------------------------------------ | -------- | -------------- |
|      type | `default` \| `primary` \| `success` \| `warning` \| `error` \| `default` | 基本类型 |
|      size | `normal`, `large`, `small`                                               | `normal` | 尺寸           |

|     style | `React.CSSProperties`                                                    | -        | 样式           |

## 写个组件

import { useState } from 'react';

export const C = () => {
  const [count, setCount] = useState(0);
  return (
    <button
      className="px-3 py-1 rounded border border-blue-500 text-blue-500 hover:bg-blue-500 hover:text-white transition-colors"
      onClick={() => setCount(count + 1)}
    >
      Hello World {count}
    </button>
  );
};

<C />
